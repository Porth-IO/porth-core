113 [[nodiscard]] auto push(const PorthDescriptor& desc) noexcept -> bool {
114 const uint32_t h = m_head.load(std::memory_order_relaxed);
115 const uint32_t t = m_tail.load(std::memory_order_acquire);
116
117 // Check if full: next head would hit tail
118 if (((h + 1) & (SIZE - 1)) == t) {
119 return false;
120 }
121
122 // Write the data to the current head slot
123 m_buffer[h] = desc;
124
125 // Release: Ensures buffer write is visible before the head index moves
126 m_head.store((h + 1) & (SIZE - 1), std::memory_order_release);
127 return true;
128 }
129
137 [[nodiscard]] auto pop(PorthDescriptor& out_desc) noexcept -> bool {
138 const uint32_t t = m_tail.load(std::memory_order_relaxed);
139 const uint32_t h = m_head.load(std::memory_order_acquire);
140
141 // Check if empty: tail caught up to head
142 if (h == t) {
143 return false;
144 }
145
146 // Read the data from the current tail slot
147 out_desc = m_buffer[t];
148
149 // Release: Signals to producer that this slot is now free for new data
150 m_tail.store((t + 1) & (SIZE - 1), std::memory_order_release);
151 return true;
152 }
