72template <size_t SIZE = DEFAULT_RING_CAPACITY>
73class PorthRingBuffer {
74 // Bitwise Wrap-around Guard: Ensures (index & (SIZE - 1)) works as a modulo.
75 static_assert((SIZE & (SIZE - 1)) == 0,
76 "SIZE must be a power of two to avoid costly division instructions.");
77
78private:
83 gsl::owner<PorthDescriptor*> m_buffer;
84 bool m_owns_buffer;

86 // Cache-line 1: The Producer's territory (Typically the Chip or TX side).
87 // Aligned to 64 bytes to ensure the producer thread owns this cache line exclusively.
88 alignas(RING_CACHE_LINE_SIZE) std::atomic<uint32_t> m_head{0};
89 std::array<uint8_t, RING_CACHE_LINE_SIZE - sizeof(std::atomic<uint32_t>)> m_pad0{};
90
91 // Cache-line 2: The Consumer's territory (Typically the Driver or RX side).
92 // Isolated to prevent the consumer's 'tail' updates from invalidating the producer's L1 cache.
93 alignas(RING_CACHE_LINE_SIZE) std::atomic<uint32_t> m_tail{0};
94 std::array<uint8_t, RING_CACHE_LINE_SIZE - sizeof(std::atomic<uint32_t>)> m_pad1{};

103 explicit PorthRingBuffer(PorthDescriptor* external_buffer = nullptr)
104 // NOLINTNEXTLINE(cppcoreguidelines-owning-memory)

129 [[nodiscard]] auto push(const PorthDescriptor& desc) noexcept -> bool {
130 const uint32_t h = m_head.load(std::memory_order_relaxed);
131 const uint32_t t = m_tail.load(std::memory_order_acquire);
132
133 // Check if full: next head would hit tail.
134 // The bitwise AND is used here as a zero-cost modulo.
135 if (((h + 1) & (SIZE - 1)) == t) {
136 return false;
137 }
138
139 // Write the data to the current head slot.
140 m_buffer[h] = desc;
141
142 // Release: Ensures all previous memory writes are visible to the consumer core.
143 m_head.store((h + 1) & (SIZE - 1), std::memory_order_release);
144 return true;
145 }
146
155 [[nodiscard]] auto pop(PorthDescriptor& out_desc) noexcept -> bool {
156 const uint32_t t = m_tail.load(std::memory_order_relaxed);
157 const uint32_t h = m_head.load(std::memory_order_acquire);
158
159 // Check if empty: tail caught up to head.
160 if (h == t) {
161 return false;
162 }
163
164 // Read the data from the current tail slot.
165 out_desc = m_buffer[t];
166
167 // Release: Signals to the producer core that this slot is now logically free.
168 m_tail.store((t + 1) & (SIZE - 1), std::memory_order_release);
169 return true;
170 }
171

178 PorthRingBuffer(PorthRingBuffer&&) = delete;
179 auto operator=(PorthRingBuffer&&) -> PorthRingBuffer& = delete;
