80template <size_t SIZE = DEFAULT_RING_CAPACITY>
81class PorthRingBuffer {
82 // Bitwise Wrap-around Guard: Ensures (index & (SIZE - 1)) works as a modulo.
83 static_assert((SIZE & (SIZE - 1)) == 0,
84 "SIZE must be a power of two to avoid costly division instructions.");
85
86private:
91 gsl::owner<PorthDescriptor*> m_buffer = nullptr;
92 bool m_owns_buffer = false;
93
94 // Cache-line 1: The Producer's territory (Typically the Chip or TX side).
95 // Aligned to 64 bytes to ensure the producer thread owns this cache line exclusively.
96 alignas(RING_CACHE_LINE_SIZE) std::atomic<uint32_t> m_head{0};
97 std::array<uint8_t, RING_CACHE_LINE_SIZE - sizeof(std::atomic<uint32_t>)> m_pad0{};
98
99 // Cache-line 2: The Consumer's territory (Typically the Driver or RX side).
100 // Isolated to prevent the consumer's 'tail' updates from invalidating the producer's L1 cache.
101 alignas(RING_CACHE_LINE_SIZE) std::atomic<uint32_t> m_tail{0};
102 std::array<uint8_t, RING_CACHE_LINE_SIZE - sizeof(std::atomic<uint32_t>)> m_pad1{};
103
104 // Ensure the rest of the class doesn't bleed into the tail's cache line
105 alignas(RING_CACHE_LINE_SIZE) std::array<char, 0> m_final_pad{};
106

114 explicit PorthRingBuffer(PorthDescriptor* external_buffer = nullptr)
115 // NOLINTNEXTLINE(cppcoreguidelines-owning-memory)
116 : m_buffer(external_buffer), m_owns_buffer(external_buffer == nullptr) {
117 if (m_owns_buffer) {
118 m_buffer = new PorthDescriptor[SIZE];

120 }

140 [[nodiscard]] auto push(const PorthDescriptor& desc) noexcept -> bool {
141 const uint32_t h = m_head.load(std::memory_order_relaxed);
142 const uint32_t t = m_tail.load(std::memory_order_acquire);
143
144 // Check if full: next head would hit tail.
145 // The bitwise AND is used here as a zero-cost modulo.
146 if (((h + 1) & (SIZE - 1)) == t) {
147 return false;
148 }
149
150 // Write the data to the current head slot.
151 m_buffer[h] = desc;
152
153 // Release: Ensures all previous memory writes are visible to the consumer core.
154 m_head.store((h + 1) & (SIZE - 1), std::memory_order_release);
155 return true;
156 }
157
166 [[nodiscard]] auto pop(PorthDescriptor& out_desc) noexcept -> bool {
167 const uint32_t t = m_tail.load(std::memory_order_relaxed);
168 const uint32_t h = m_head.load(std::memory_order_acquire);
169
170 // Check if empty: tail caught up to head.
171 if (h == t) {
172 return false;
173 }
174
175 // Read the data from the current tail slot.
176 out_desc = m_buffer[t];
177
178 // Release: Signals to the producer core that this slot is now logically free.
179 m_tail.store((t + 1) & (SIZE - 1), std::memory_order_release);
180 return true;
181 }

185 auto operator=(const PorthRingBuffer&) -> PorthRingBuffer& = delete;

187 // Moving is deleted to ensure the physical memory address remains stable

190 auto operator=(PorthRingBuffer&&) -> PorthRingBuffer& = delete;
