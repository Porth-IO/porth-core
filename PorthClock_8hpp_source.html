70 static auto now_precise() noexcept -> uint64_t {
71#if defined(__x86_64__) || defined(__i386__)
72 unsigned int aux = 0;
73 // RDTSCP is a serializing instruction; it forces the CPU to wait until
74 // all previous instructions in the stream have executed.
75 return __rdtscp(&aux);
76#elif defined(__aarch64__)
77 uint64_t val = 0;
78 // ISB (Instruction Synchronization Barrier) flushes the pipeline and
79 // ensures that the MRS read is not speculatively executed ahead of
80 // previous logic. This is the ARM equivalent of x86 serialization.
81 asm volatile("isb; mrs %0, cntvct_el0" : "=r"(val));
82 return val;
83#endif
