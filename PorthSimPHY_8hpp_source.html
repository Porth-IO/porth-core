88 auto apply_protocol_delay() noexcept -> void {
89 // Task 4.3: Calculate thermal jitter multiplier
90 // Jitter increases by 1ns for every 1 degree above 40C (40000 mC)
91 uint64_t thermal_jitter = 0;
92 const uint32_t temp = current_temp_mc.load(std::memory_order_relaxed);
93 if (temp > THERMAL_THRESHOLD_MC) {
94 thermal_jitter = (temp - THERMAL_THRESHOLD_MC) / MC_TO_C_DIVISOR;
95 }
96
97 const int64_t current_jitter = (jitter_ns > 0) ? jitter_dist(gen) : 0;
98 uint64_t total_delay_ns =
99 base_delay_ns + static_cast<uint64_t>(current_jitter) + fec_penalty_ns + thermal_jitter;
100
101 // Simulate FEC retry spike (Task 2.2)
102 if (error_dist(gen) < fec_error_rate) {
103 total_delay_ns += FEC_RETRY_SPIKE_NS;
104 }
105

107 static_cast<uint64_t>(static_cast<double>(total_delay_ns) * cycles_per_ns);
108 const uint64_t start = PorthClock::now_precise();
