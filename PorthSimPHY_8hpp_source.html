58 [[nodiscard]] auto calculate_thermal_jitter() const noexcept -> uint64_t {
59 uint64_t thermal_jitter = 0;
60 const uint32_t temp = m_current_temp_mc.load(std::memory_order_relaxed);
61
62 if (temp > THERMAL_THRESHOLD_MC) {
63 thermal_jitter = (temp - THERMAL_THRESHOLD_MC) / MC_TO_C_DIVISOR;
64 }
65 return thermal_jitter;
66 }

69 [[nodiscard]] auto get_fec_penalty() noexcept -> uint64_t {
70 if (m_error_dist(m_gen) < m_fec_error_rate) {
71 return FEC_RETRY_SPIKE_NS;
72 }
73 return 0;

113 auto apply_protocol_delay() noexcept -> void {
114 // 1. Accumulate total propagation delay
115 const int64_t random_jitter = (m_jitter_ns > 0) ? m_jitter_dist(m_gen) : 0;
116
117 uint64_t total_delay_ns =
118 m_base_delay_ns + m_fec_penalty_ns + static_cast<uint64_t>(random_jitter);
119
120 // 2. Add physics-based penalties
121 total_delay_ns += calculate_thermal_jitter();
122 total_delay_ns += get_fec_penalty();
123
124 // 3. Convert to cycles and execute high-precision wait
125 const auto target_cycles =
126 static_cast<uint64_t>(static_cast<double>(total_delay_ns) * m_cycles_per_ns);
127
128 const uint64_t start_cycles = PorthClock::now_precise();
129 while (PorthClock::now_precise() - start_cycles < target_cycles) {
130 cpu_relax();
131 }
132 }
