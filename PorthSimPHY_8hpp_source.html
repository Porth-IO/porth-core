103 [[nodiscard]] auto calculate_thermal_jitter() const noexcept -> uint64_t {
104 uint64_t thermal_jitter = 0;
105 const uint32_t temp = m_current_temp_mc.load(std::memory_order_relaxed);
106
107 if (temp > THERMAL_THRESHOLD_MC) {
108 // Linear approximation of jitter increase per degree Celsius above threshold.
109 thermal_jitter = (temp - THERMAL_THRESHOLD_MC) / MC_TO_C_DIVISOR;
110 }
111 return thermal_jitter;
112 }

118 [[nodiscard]] auto get_fec_penalty(int32_t current_snr) const noexcept -> uint64_t {
119 // As SNR (from PorthDeviceLayout) drops, the probability of an FEC retry increases
120 // exponentially. Standard SNR is 30.00 dB (represented as 3000 in the register)
121 double error_rate = m_fec_error_rate;
122 if (current_snr < SNR_CRITICAL_THRESHOLD) {
123 error_rate *= FEC_LOW_SNR_MULTIPLIER; // 10x error increase at low SNR
124 }
125
126 if (m_error_dist(m_gen) < error_rate) {
127 return FEC_RETRY_SPIKE_NS;
128 }
129 return 0;
130 }

137 static void cpu_relax() noexcept {
138#if defined(__i386__) || defined(__x86_64__)
139 // PAUSE: Notifies the CPU that we are in a spin-loop, improving power
140 // efficiency and reducing the exit-latency of the loop.
141 asm volatile("pause" ::: "memory");
142#elif defined(__aarch64__)
143 // ISB: Flushes the pipeline on ARM64 to ensure the loop condition is
144 // re-evaluated without speculative interference.
145 asm volatile("isb" ::: "memory");
146#endif

169 auto update_thermal_load(uint32_t temp_mc) noexcept -> void {
170 // Thermal Inertia: Instead of an instant "jump", we model the heat capacity
171 // of the InP lattice. The temperature "sloughs" towards the target.
172 const uint32_t current = m_current_temp_mc.load(std::memory_order_relaxed);
173 if (temp_mc > current) {
174 m_current_temp_mc.store(current + HEATING_STEP_MC,
175 std::memory_order_relaxed); // Gradual heating
176 } else if (temp_mc < current) {
177 m_current_temp_mc.store(current - COOLING_STEP_MC,
178 std::memory_order_relaxed); // Gradual cooling
179 }
180 }

196 auto apply_protocol_delay(int32_t current_snr = STANDARD_SNR_DB) noexcept -> void {
197 // 1. Accumulate total propagation delay from the physics model.
198 int64_t random_jitter = 0;
199 if (m_jitter_ns > 0) {
200 random_jitter = m_jitter_dist(m_gen);
201 }
202
203 uint64_t total_delay_ns =
204 m_base_delay_ns + m_fec_penalty_ns + static_cast<uint64_t>(random_jitter);
205
206 // 2. Add non-deterministic penalties (Physics-based).
207 total_delay_ns += calculate_thermal_jitter();
208 total_delay_ns += get_fec_penalty(current_snr);
209
210 // 3. Convert time to CPU cycles for high-precision wait.
211 const auto target_cycles =
212 static_cast<uint64_t>(static_cast<double>(total_delay_ns) * m_cycles_per_ns);
213
214 // Sovereign Spin-Loop: Guarantees timing precision to within ~3-5 cycles.
215 const uint64_t start_cycles = PorthClock::now_precise();
216 while (PorthClock::now_precise() - start_cycles < target_cycles) {
217 cpu_relax();
218 }
219 }
