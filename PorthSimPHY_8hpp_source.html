77 [[nodiscard]] auto calculate_thermal_jitter() const noexcept -> uint64_t {
78 uint64_t thermal_jitter = 0;
79 const uint32_t temp = m_current_temp_mc.load(std::memory_order_relaxed);
80
81 if (temp > THERMAL_THRESHOLD_MC) {
82 // Linear approximation of jitter increase per degree Celsius above threshold.
83 thermal_jitter = (temp - THERMAL_THRESHOLD_MC) / MC_TO_C_DIVISOR;
84 }
85 return thermal_jitter;
86 }

103 static void cpu_relax() noexcept {
104#if defined(__i386__) || defined(__x86_64__)
105 // PAUSE: Notifies the CPU that we are in a spin-loop, improving power
106 // efficiency and reducing the exit-latency of the loop.
107 asm volatile("pause" ::: "memory");
108#elif defined(__aarch64__)
109 // ISB: Flushes the pipeline on ARM64 to ensure the loop condition is
110 // re-evaluated without speculative interference.
111 asm volatile("isb" ::: "memory");

147 auto apply_protocol_delay() noexcept -> void {
148 // 1. Accumulate total propagation delay from the physics model.
149 const int64_t random_jitter = (m_jitter_ns > 0) ? m_jitter_dist(m_gen) : 0;
150
151 uint64_t total_delay_ns =
152 m_base_delay_ns + m_fec_penalty_ns + static_cast<uint64_t>(random_jitter);
153
154 // 2. Add non-deterministic penalties (Physics-based).
155 total_delay_ns += calculate_thermal_jitter();
156 total_delay_ns += get_fec_penalty();
157
158 // 3. Convert time to CPU cycles for high-precision wait.
159 const auto target_cycles =
160 static_cast<uint64_t>(static_cast<double>(total_delay_ns) * m_cycles_per_ns);
161
162 // Sovereign Spin-Loop: Guarantees timing precision to within ~3-5 cycles.
163 const uint64_t start_cycles = PorthClock::now_precise();
164 while (PorthClock::now_precise() - start_cycles < target_cycles) {
165 cpu_relax();
166 }
167 }
