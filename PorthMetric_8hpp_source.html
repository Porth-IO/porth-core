70 [[nodiscard]] auto calculate_mean(const std::vector<uint64_t>& samples) const noexcept

81 [[nodiscard]] auto calculate_stdev(const std::vector<uint64_t>& samples,
82 double mean) const noexcept -> double {
83 const double sq_sum =
84 std::inner_product(samples.begin(), samples.end(), samples.begin(), 0.0);
85 return std::sqrt(std::abs((sq_sum / static_cast<double>(m_count)) - (mean * mean)));

185 std::cout << std::format("StDev: {:.2f} ns\n", stdev_cycles / cycles_per_ns);
186 std::cout << std::format("IQR: {:.2f} ns\n", q3_ns - q1_ns); // Measures consistency
187 std::cout << std::format("P99.99: {:.2f} ns\n", p99_ns); // Measures worst-case jitter
