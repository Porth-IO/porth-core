49 [[nodiscard]] auto get_sorted_samples() const -> std::vector<uint64_t> {
50 std::vector<uint64_t> sorted(m_samples.begin(),

57 [[nodiscard]] auto calculate_mean(const std::vector<uint64_t>& samples) const noexcept
58 -> double {
59 const double sum = std::accumulate(samples.begin(), samples.end(), 0.0);
60 return sum / static_cast<double>(m_count);
61 }

64 [[nodiscard]] auto calculate_stdev(const std::vector<uint64_t>& samples,
65 double mean) const noexcept -> double {
66 const double sq_sum =
67 std::inner_product(samples.begin(), samples.end(), samples.begin(), 0.0);
68 return std::sqrt(std::abs((sq_sum / static_cast<double>(m_count)) - (mean * mean)));
69 }

72 [[nodiscard]] auto get_percentile_ns(const std::vector<uint64_t>& samples,
73 double percentile,
74 double cpns) const noexcept -> double {
75 auto idx = static_cast<size_t>(percentile * static_cast<double>(m_count) / 100.0);
76 if (idx >= m_count) {
77 idx = m_count - 1;
78 }
79 return static_cast<double>(samples[idx]) / cpns;
80 }

142 void print_stats(double cycles_per_ns) {
143 if (m_count == 0)
144 return;
145
146 // 1. Prepare Data
147 const auto sorted_samples = get_sorted_samples();
148
149 // 2. Compute Statistics (in cycles)
150 const double mean_cycles = calculate_mean(sorted_samples);
151 const double stdev_cycles = calculate_stdev(sorted_samples, mean_cycles);
152
153 // 3. Compute Percentiles (converted to ns)
154 const double q1_ns = get_percentile_ns(sorted_samples, PERCENTILE_Q1, cycles_per_ns);
155 const double q3_ns = get_percentile_ns(sorted_samples, PERCENTILE_Q3, cycles_per_ns);
156 const double p99_ns = get_percentile_ns(sorted_samples, PERCENTILE_P99_99, cycles_per_ns);
157
158 // 4. Final Jitter Analysis Output
159 std::cout << "\n--- Porth-IO Jitter Analysis (ns) ---\n";
160 std::cout << std::format("Mean: {:.2f} ns\n", mean_cycles / cycles_per_ns);
161 std::cout << std::format("StDev: {:.2f} ns\n", stdev_cycles / cycles_per_ns);
162 std::cout << std::format("IQR: {:.2f} ns\n", q3_ns - q1_ns);
163 std::cout << std::format("P99.99: {:.2f} ns\n", p99_ns);
164 }
