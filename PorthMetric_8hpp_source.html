95 // Copy and sort for percentile analysis

100 // Basic statistics
101 const double sum = std::accumulate(sorted_samples.begin(), sorted_samples.end(), 0.0);
102 const double mean = sum / static_cast<double>(count);
103
104 // Standard Deviation using inner product for precision
105 const double sq_sum = std::inner_product(
106 sorted_samples.begin(), sorted_samples.end(), sorted_samples.begin(), 0.0);
107 const double stdev =
108 std::sqrt(std::abs((sq_sum / static_cast<double>(count)) - (mean * mean)));
109
110 auto get_p = [&](double percentile) {
111 auto idx = static_cast<size_t>(percentile * static_cast<double>(count) / 100.0);
112 if (idx >= count) {
113 idx = count - 1;
114 }
115 return static_cast<double>(sorted_samples[idx]) / cycles_per_ns;
116 };
117
118 const double q1 = get_p(PERCENTILE_Q1);
119 const double q3 = get_p(PERCENTILE_Q3);
120 const double iqr = q3 - q1;
121
122 // Immaculate C++23 formatted output
123 std::cout << "\n--- Porth-IO Jitter Analysis (ns) ---\n";
124 std::cout << std::format("Mean: {:.2f} ns\n", mean / cycles_per_ns);
125 std::cout << std::format("StDev: {:.2f} ns\n", stdev / cycles_per_ns);
126 std::cout << std::format("IQR: {:.2f} ns\n", iqr);
127 std::cout << std::format("P99.99: {:.2f} ns\n", get_p(PERCENTILE_P99_99));
