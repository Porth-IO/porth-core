86 [[nodiscard]] auto calculate_stdev(const std::vector<uint64_t>& samples,

88 const double sq_sum =

178 void print_stats(double cycles_per_ns) {

183 const auto sorted_samples = get_sorted_samples();

190 const double p99_ns = get_percentile_ns(sorted_samples, PERCENTILE_P99_99, cycles_per_ns);
191
192 std::cout << "\n--- Porth-IO Jitter Analysis (ns) ---\n";
193 std::cout << std::format("Mean: {:.2f} ns\n", mean_cycles / cycles_per_ns);
194 std::cout << std::format("StDev: {:.2f} ns\n", stdev_cycles / cycles_per_ns);
195 std::cout << std::format("IQR: {:.2f} ns\n", q3_ns - q1_ns); // Measures consistency
196 std::cout << std::format("P99.99: {:.2f} ns\n", p99_ns); // Measures worst-case jitter
197 }
