81 void run_physics_loop() { // NOLINT(readability-function-cognitive-complexity)
87 (void)pin_thread_to_core(0);
88
89 PorthDeviceLayout* dev = mock_hw.view();
90
91 uint32_t temp = SIM_BASE_TEMP_MC; // Base operating temperature of 25.0 C (milli-Celsius)
92 std::mt19937 gen(std::random_device{}());
93 std::uniform_int_distribution<uint32_t> bit_dist(0, SIM_STATUS_MAX_BIT);
94
95 while (run_sim.load(std::memory_order_relaxed)) {
96 // Signal to the driver that hardware logic is active and ready
97 dev->status.write(0x1);
98 // Use release semantics to ensure the write is visible to other cores
99 std::atomic_thread_fence(std::memory_order_release);
100
101 if (!inject_deadlock.load(std::memory_order_relaxed)) {
102
107 uint64_t shuttle_addr = dev->data_ptr.load();
108 if (shuttle_addr != 0) {
109 // NOLINTNEXTLINE(performance-no-int-to-ptr)
110 auto* shuttle =
111 reinterpret_cast<PorthShuttle<SIM_DEFAULT_SHUTTLE_SIZE>*>(shuttle_addr);
112 PorthDescriptor desc{};
113 // Drain the shuttle as fast as the "hardware" can process it
114 while (shuttle->ring()->pop(desc)) {
115 // In a real device, this is where photons are fired.
116 }
117 }
118
119 // Task 3: Photonics/GaN Model
120 // Heating logic: Increases temp by 0.1C when active (control == 0x1)
121 if (dev->control.load() == 0x1) {
122 temp += SIM_TEMP_INC_MC;
123 } else {
124 // Cooling logic: Decreases temp by 0.05C until base temp reached
125 if (temp > SIM_BASE_TEMP_MC) {
126 temp -= SIM_TEMP_DEC_MC;
127 }
128 }
129
130 dev->laser_temp.write(temp);
131
132 // Task 4.3: Thermal Jitter Feedback
133 // Pushes current temperature back into the PHY delay engine to calculate jitter
134 phy.update_thermal_load(temp);
135
136 // GaN Power Rail Noise (Simulating 48V base + millivolt-scale jitter)
137 [[maybe_unused]] const uint32_t noise = (static_cast<uint32_t>(std::rand()) % 100);
138 // Note: gan_voltage is modeled in the extended register map
139 // dev->gan_voltage.write(48000 + noise);
140
141 // Sub-task 4.1: Register Corruption (Single-event upset emulation)
142 if (corrupt_status.load(std::memory_order_relaxed)) {
143 uint32_t current_status = dev->status.load();
144 dev->status.write(current_status ^ (1U << bit_dist(gen)));
145 }
146 }
147 std::this_thread::sleep_for(std::chrono::microseconds(SIM_PHYSICS_STEP_US));
148 }
