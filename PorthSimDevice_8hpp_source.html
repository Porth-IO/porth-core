119 void update_thermal_model(PorthDeviceLayout* dev, uint32_t& current_temp) noexcept {

135 void apply_chaos_effects(PorthDeviceLayout* dev,
136 std::mt19937& gen,
137 std::uniform_int_distribution<uint32_t>& bit_dist) noexcept {
138 if (m_corrupt_status.load(std::memory_order_relaxed)) {
139 const uint32_t current_status = dev->status.load();
140
141 // 10% probability of critical DMA pointer corruption vs status bit-flip.
142 if (bit_dist(gen) > SIM_CHAOS_THRESHOLD) {
143 dev->data_ptr.write(dev->data_ptr.load() ^ (1ULL << bit_dist(gen)));
144 } else {
145 dev->status.write(current_status ^ (1U << bit_dist(gen)));

155 void run_physics_loop() {
156 // Technical Justification: Pinning to Core 0 ensures the physics steps
157 // (100us) remain deterministic and are not preempted by general OS tasks.
158 if (!pin_thread_to_core(0)) {
159 // HFT Best Practice: If pinning fails, the simulation fidelity is compromised.
160 std::cerr << "[Porth-Sim] Warning: Failed to pin physics thread to Core 0.\n";
161 }
162
163 PorthDeviceLayout* dev = m_mock_hw.view();
164 uint32_t temp = SIM_BASE_TEMP_MC;
165
166 std::mt19937 gen(std::random_device{}());
167 std::uniform_int_distribution<uint32_t> bit_dist(0, SIM_STATUS_MAX_BIT);
168
169 while (m_run_sim.load(std::memory_order_relaxed)) {
170 // Signal "Hardware Ready" to the Logic Layer.
171 dev->status.write(0x1);
172 std::atomic_thread_fence(std::memory_order_release);
173
174 if (!m_inject_deadlock.load(std::memory_order_relaxed)) {
175 process_dma(dev);
176 update_thermal_model(dev, temp);
177 apply_chaos_effects(dev, gen, bit_dist);
178 }
179
180 std::this_thread::sleep_for(std::chrono::microseconds(SIM_PHYSICS_STEP_US));
181 }
