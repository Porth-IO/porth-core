79 void process_dma(PorthDeviceLayout* dev) noexcept {
80 const uint64_t shuttle_addr = dev->data_ptr.load();
81 if (shuttle_addr == 0)
82 return;

85 if (expected != 0 && shuttle_addr != expected) {

103 void update_thermal_model(PorthDeviceLayout* dev, uint32_t& current_temp) noexcept {
104 // Heating logic: Increases temp when active (control == 0x1)
105 if (dev->control.load() == 0x1) {
106 current_temp += SIM_TEMP_INC_MC;
107 } else if (current_temp > SIM_BASE_TEMP_MC) {
108 // Cooling logic: Decreases temp until base reached
109 current_temp -= SIM_TEMP_DEC_MC;
110 }
111
112 dev->laser_temp.write(current_temp);
113 m_phy.update_thermal_load(current_temp);
114 }

117 void apply_chaos_effects(PorthDeviceLayout* dev,
118 std::mt19937& gen,
119 std::uniform_int_distribution<uint32_t>& bit_dist) noexcept {
120 if (m_corrupt_status.load(std::memory_order_relaxed)) {
121 const uint32_t current_status = dev->status.load();
122
123 // Randomly flip bits in status OR corrupt the data_ptr
124 // to simulate a hard DMA pointer failure (approx 10% chance).
125 if (bit_dist(gen) > 28) {
126 dev->data_ptr.write(dev->data_ptr.load() ^ (1ULL << bit_dist(gen)));
127 } else {
128 dev->status.write(current_status ^ (1U << bit_dist(gen)));
129 }
130 }

137 void run_physics_loop() {
138 (void)pin_thread_to_core(0);
139
140 PorthDeviceLayout* dev = m_mock_hw.view();
141 uint32_t temp = SIM_BASE_TEMP_MC;
142
143 std::mt19937 gen(std::random_device{}());
144 std::uniform_int_distribution<uint32_t> bit_dist(0, SIM_STATUS_MAX_BIT);
145
146 while (m_run_sim.load(std::memory_order_relaxed)) {
147 // Hardware Ready Signal
148 dev->status.write(0x1);
149 std::atomic_thread_fence(std::memory_order_release);
150
151 // Execute physics steps if not deadlocked
152 if (!m_inject_deadlock.load(std::memory_order_relaxed)) {
153 process_dma(dev);
154 update_thermal_model(dev, temp);
155 apply_chaos_effects(dev, gen, bit_dist);
156 }
157
158 std::this_thread::sleep_for(std::chrono::microseconds(SIM_PHYSICS_STEP_US));
159 }
160 }
