96 void process_dma(PorthDeviceLayout* dev) noexcept {
97 const uint64_t shuttle_addr = dev->data_ptr.load();
98 if (shuttle_addr == 0) {
99 return;
100 }
101
102 // Chaos Guard: Ensure address is 64-byte aligned.
103 if ((shuttle_addr & SimulationPhysics::SHUTTLE_ALIGN_MASK) != 0) {
104 return;
105 }
106
107 uint64_t expected = m_last_valid_shuttle.load();
108 if (expected != 0 && shuttle_addr != expected) {
109 return;
110 }
111
112 if (expected == 0) {
113 m_last_valid_shuttle.store(shuttle_addr);
114 }
115
116 auto* ring = std::bit_cast<PorthRingBuffer<SIM_DEFAULT_SHUTTLE_SIZE>*>(shuttle_addr);
117 PorthDescriptor desc{};
118
119 while (ring->pop(desc)) {
120 dev->counter.write(dev->counter.load() + 1);
121 }
122 }

125 void update_thermal_model(PorthDeviceLayout* dev, uint32_t& current_temp) noexcept {
126 uint32_t volt = dev->gan_voltage.load();
127 if (dev->control.load() == 0x1) {
128 if (volt < SimulationPhysics::GAN_VOLT_LIMIT_UV) {
129 volt += SimulationPhysics::GAN_VOLT_STEP_UV;
130 }
131 } else {
132 if (volt > 0) {
133 volt = (volt > SimulationPhysics::GAN_VOLT_COOL_UV)
134 ? volt - SimulationPhysics::GAN_VOLT_COOL_UV
135 : 0;
136 }
137 }
138 dev->gan_voltage.write(volt);
139
140 uint32_t target_temp = SIM_BASE_TEMP_MC;
141 if (volt > 0) {
142 target_temp +=
143 (volt / SimulationPhysics::VOLT_TEMP_DIV) * SimulationPhysics::VOLT_TEMP_MULT;
144 }
145
146 m_phy.update_thermal_load(target_temp);
147 current_temp = m_phy.get_current_temp();
148
149 int32_t snr = SimulationPhysics::SNR_BASELINE_MC;
150 if (current_temp > THERMAL_THRESHOLD_MC) {
151 snr -= static_cast<int32_t>((current_temp - THERMAL_THRESHOLD_MC) /
152 SimulationPhysics::SNR_TEMP_DIV);
153 }
154 dev->rf_snr.write(snr);
155 dev->laser_temp.write(current_temp);
156 }

158 void apply_chaos_effects(PorthDeviceLayout* dev,
159 std::mt19937& gen,
160 std::uniform_int_distribution<uint32_t>& bit_dist) noexcept {
161 if (m_corrupt_status.load(std::memory_order_relaxed)) {
162 const uint32_t current_status = dev->status.load();
163 if (bit_dist(gen) > SIM_CHAOS_THRESHOLD) {
164 dev->data_ptr.write(dev->data_ptr.load() ^ (1ULL << bit_dist(gen)));

171 void run_physics_loop() {
172 if (!pin_thread_to_core(0)) {
173 std::cerr << "[Porth-Sim] Warning: Failed to pin physics thread to Core 0.\n";
174 }
175
176 PorthDeviceLayout* dev = m_mock_hw.view();
177 uint32_t temp = SIM_BASE_TEMP_MC;
178
179 std::mt19937 gen(std::random_device{}());
180 std::uniform_int_distribution<uint32_t> bit_dist(0, SIM_STATUS_MAX_BIT);
181
182 while (m_run_sim.load(std::memory_order_relaxed)) {
183 dev->status.write(0x1);
184 std::atomic_thread_fence(std::memory_order_release);
185
186 if (!m_inject_deadlock.load(std::memory_order_relaxed)) {
187 process_dma(dev);
188 update_thermal_model(dev, temp);
189 apply_chaos_effects(dev, gen, bit_dist);
190 }
191 std::this_thread::sleep_for(std::chrono::microseconds(SIM_PHYSICS_STEP_US));
192 }
