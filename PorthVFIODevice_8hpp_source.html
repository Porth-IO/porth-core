61 explicit PorthVFIODevice(const std::string& pci_addr) {
62 // 1. Open the VFIO Container
63 m_container_fd = open("/dev/vfio/vfio", O_RDWR);
64 if (m_container_fd < 0) {
65 throw std::runtime_error(
66 "Failed to open /dev/vfio/vfio. Check permissions (sudo required).");
67 }
68
69 if (ioctl(m_container_fd, VFIO_GET_API_VERSION) != VFIO_API_VERSION) {
70 throw std::runtime_error("Incompatible VFIO API version.");
71 }
72
73 // 2. Open the IOMMU Group
74 int group_id = get_iommu_group(pci_addr);
75 m_group_fd = open(std::format("/dev/vfio/{}", group_id).c_str(), O_RDWR);
76 if (m_group_fd < 0) {
77 throw std::runtime_error(std::format("Failed to open IOMMU group {}.", group_id));
78 }
79
80 // 3. Add Group to Container
81 if (ioctl(m_group_fd, VFIO_GROUP_SET_CONTAINER, &m_container_fd) < 0) {
82 throw std::runtime_error("Failed to set VFIO container.");
83 }
84
85 // 4. Set IOMMU Type (Type1 is standard for x86/ARM64)
86 if (ioctl(m_container_fd, VFIO_SET_IOMMU, VFIO_TYPE1_IOMMU) < 0) {
87 throw std::runtime_error("Failed to set IOMMU type to Type1.");
88 }
89
90 // 5. Get Device File Descriptor
91 m_device_fd = ioctl(m_group_fd, VFIO_GROUP_GET_DEVICE_FD, pci_addr.c_str());
92 if (m_device_fd < 0) {
93 throw std::runtime_error(std::format("Failed to get device FD for {}.", pci_addr));
94 }
95
96 // 6. Map BAR 0 (The Newport Cluster Control Region)
97 struct vfio_region_info reg{};
98 reg.argsz = sizeof(reg);
99 reg.index = VFIO_PCI_BAR0_REGION_INDEX;
100
101 if (ioctl(m_device_fd, VFIO_DEVICE_GET_REGION_INFO, &reg) < 0) {
102 throw std::runtime_error("Failed to get BAR0 region info.");
103 }
104
105 m_bar_size = reg.size;
106 m_bar_ptr = mmap(nullptr,
107 m_bar_size,
108 PROT_READ | PROT_WRITE,
109 MAP_SHARED,
110 m_device_fd,
111 static_cast<off_t>(reg.offset));
112
113 if (m_bar_ptr == MAP_FAILED) {
114 throw std::runtime_error("Failed to mmap BAR0 into userspace.");
115 }
116
117 std::cout << std::format(
118 "[Porth-VFIO] Sovereign Ownership Established: {} | BAR0 at {}\n", pci_addr, m_bar_ptr);
119 }

138 auto map_dma(void* vaddr, size_t size) -> uint64_t {
139 struct vfio_iommu_type1_dma_map dma_map{};
140 dma_map.argsz = sizeof(dma_map);
141 dma_map.flags = VFIO_DMA_MAP_FLAG_READ | VFIO_DMA_MAP_FLAG_WRITE;
142 dma_map.vaddr = reinterpret_cast<uintptr_t>(vaddr);
143 dma_map.iova = reinterpret_cast<uintptr_t>(vaddr); // 1:1 mapping for simplicity
144 dma_map.size = size;
145
146 if (ioctl(m_container_fd, VFIO_IOMMU_MAP_DMA, &dma_map) < 0) {
147 throw std::runtime_error("Failed to map DMA region in IOMMU.");
148 }
149
150 std::cout << std::format("[Porth-VFIO] DMA Region Mapped: VADDR {} -> IOVA {}\n",
151 vaddr,
152 dma_map.iova)
153 << std::flush;
154 return dma_map.iova;
155 }
