149 auto map_dma(void* vaddr, size_t size) const -> uint64_t {
150 struct vfio_iommu_type1_dma_map dma_map{};
151 dma_map.argsz = sizeof(dma_map);
152 dma_map.flags = VFIO_DMA_MAP_FLAG_READ | VFIO_DMA_MAP_FLAG_WRITE;
153 // NOLINTBEGIN(cppcoreguidelines-pro-type-reinterpret-cast)
154 dma_map.vaddr = reinterpret_cast<uintptr_t>(vaddr);
155 dma_map.iova = reinterpret_cast<uintptr_t>(vaddr); // 1:1 mapping for simplicity
156 // NOLINTEND(cppcoreguidelines-pro-type-reinterpret-cast)
157 dma_map.size = size;
158
159 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-vararg)
160 if (ioctl(m_container_fd, VFIO_IOMMU_MAP_DMA, &dma_map) < 0) {
161 throw std::runtime_error("Failed to map DMA region in IOMMU.");
162 }
163
164 std::cout << std::format("[Porth-VFIO] DMA Region Mapped: VADDR {} -> IOVA {}\n",
165 vaddr,
166 dma_map.iova)
167 << std::flush;
168 return dma_map.iova;
169 }
