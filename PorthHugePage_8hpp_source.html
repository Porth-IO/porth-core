44class PorthHugePage {
45private:
46 gsl::owner<void*> m_ptr = nullptr;
47 size_t m_total_size;
48 int m_node;
49 bool m_is_numa_managed{false};
50 bool m_is_mmaped = false;
51
52 static constexpr size_t HP_SIZE = static_cast<size_t>(2) * 1024 * 1024;
53
55 static constexpr size_t BITS_PER_BYTE = 8;
56
58 void allocate_initial_buffer() {
59 // Verify NUMA node is available before allocation
60 if (numa_available() < 0) {
61 std::cerr << "[Porth-IO] Warning: NUMA support not available in kernel.\n";
62 m_ptr = std::aligned_alloc(HP_SIZE, m_total_size);
63 m_is_numa_managed = false;
64 return;
65 }
66
67 // Strict allocation on the target node.
68 // This ensures the logic layer and Newport hardware share the same local memory bus.
69 // NOLINTNEXTLINE(cppcoreguidelines-owning-memory)
70 m_ptr = numa_alloc_onnode(m_total_size, m_node);
71
72 if (m_ptr == nullptr) {
73 std::cerr << std::format("[Porth-IO] Critical: Allocation failed on NUMA node {}.\n",
74 m_node);
75 m_ptr = std::aligned_alloc(HP_SIZE, m_total_size);
76 m_is_numa_managed = false;
77 } else {
78 m_is_numa_managed = true;
79 // Explicitly bind the memory policy to this node to prevent pages from migrating.
80 unsigned long nodemask = (1UL << m_node);
81 if (set_mempolicy(MPOL_BIND, &nodemask, (sizeof(nodemask) * BITS_PER_BYTE) + 1) != 0) {
82 std::cerr << "[Porth-IO] Warning: Could not set strict MPOL_BIND policy.\n";
83 }
84 }
85 }
86
88 void attempt_hugepage_upgrade() {
89 // MAP_LOCKED ensures the memory is pinned in RAM and never swapped to disk.
90 // MAP_POPULATE pre-faults the pages to ensure the TLB is primed before use.
91 int flags = MAP_PRIVATE | MAP_ANONYMOUS | MAP_HUGETLB | MAP_LOCKED | MAP_POPULATE;
92
93 // Declare as owner immediately so the assignment to m_ptr is valid.
94 // NOLINTNEXTLINE(cppcoreguidelines-owning-memory)
95 gsl::owner<void*> mapped_ptr =
96 mmap(m_ptr, m_total_size, PROT_READ | PROT_WRITE, flags, -1, 0);
97
98 if (mapped_ptr == MAP_FAILED) {
99 std::cout << "[Porth-IO] Note: HugePages not supported.\n";
100 m_is_mmaped = false;
101 if (mlock(m_ptr, m_total_size) != 0) {
102 if (std::getenv("CI") == nullptr) {
103 throw std::runtime_error("Porth-HugePage: mlock failed.");
104 }
105 }
106 } else {
107 if (m_is_numa_managed) {
108 numa_free(m_ptr, m_total_size);
109 } else {
110 // NOLINTNEXTLINE(cppcoreguidelines-no-malloc, cppcoreguidelines-owning-memory)
111 std::free(m_ptr);
112 }
113
114 m_ptr = mapped_ptr;
115 m_is_mmaped = true;
116 }
117 }
118
119public:
120 explicit PorthHugePage(size_t size, NumaNode numa_node = NumaNode(0))
121 : m_total_size(((size + HP_SIZE - 1) / HP_SIZE) * HP_SIZE), m_node(numa_node.value) {
122
123 allocate_initial_buffer();
124
125 if (m_ptr == nullptr) {
126 throw std::runtime_error("[Porth-IO] Fatal: Total memory allocation failure.");
127 }
128
129 attempt_hugepage_upgrade();
130
131 std::cout << std::format(
132 "[Porth-IO] Memory Initialized: {} bytes at {}\n", m_total_size, m_ptr);
133 }
134
136 ~PorthHugePage() {
137 if (m_ptr != nullptr) {
138 if (m_is_mmaped) {
139 munmap(m_ptr, m_total_size);
140 } else if (m_is_numa_managed) {
141 numa_free(m_ptr, m_total_size);
142 } else {
143 // NOLINTNEXTLINE(cppcoreguidelines-no-malloc, cppcoreguidelines-owning-memory)
144 std::free(m_ptr);
145 }
146 }
147 }
