160 template <size_t Cap>
161 void bridge_to_shuttle(PorthShuttle<Cap>& shuttle, PorthStats* stats = nullptr) noexcept {
162 uint32_t idx_rx = 0;
163 const uint32_t rcvd = xsk_ring_cons__peek(&m_rx_ring, 1, &idx_rx);
164
165 if (rcvd == 0) {
166 if (xsk_ring_prod__needs_wakeup(&m_fill_ring)) {
167 (void)recvfrom(m_xsk_fd, nullptr, 0, MSG_DONTWAIT, nullptr, nullptr);
168 }
169 return;
170 }
171
172 const struct xdp_desc* desc = xsk_ring_cons__rx_desc(&m_rx_ring, idx_rx);
173 const uint64_t packet_addr = xsk_umem__add_offset_to_addr(desc->addr);
174 const uint32_t packet_len = desc->len;
175
176 if (!is_addr_valid(packet_addr, packet_len)) {
177 if (stats != nullptr) {
178 stats->dropped_packets.fetch_add(1, std::memory_order_relaxed);
179 }
180 recycle_frame(desc->addr);
181 xsk_ring_cons__release(&m_rx_ring, 1);
182 return;
183 }
184
185 if (shuttle.ring()->push({.addr = packet_addr, .len = packet_len})) {
186 xsk_ring_cons__release(&m_rx_ring, 1);
187 if (stats != nullptr) {
188 stats->total_packets.fetch_add(1, std::memory_order_relaxed);
189 stats->total_bytes.fetch_add(packet_len, std::memory_order_relaxed);
190 }
191 recycle_frame(desc->addr);
192 } else {
193 if (stats != nullptr) {
194 stats->dropped_packets.fetch_add(1, std::memory_order_relaxed);
195 }
196 }
197 }
