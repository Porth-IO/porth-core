58[[nodiscard]] inline auto pin_thread_to_core(int core_id) noexcept -> PorthStatus {
59 cpu_set_t cpuset;
60 CPU_ZERO(&cpuset);
61 CPU_SET(core_id, &cpuset);
62
63 pthread_t current_thread = pthread_self();
64 // pthread_setaffinity_np is used to enforce hard-affinity at the pthread level.
65 int result = pthread_setaffinity_np(current_thread, sizeof(cpu_set_t), &cpuset);
66
67 if (result != 0) {
68 std::cerr << std::format("[Porth-Util] Warning: Could not pin thread to core {}\n",
69 core_id);
70 return PorthStatus::ERROR_AFFINITY;
71 }
72
73 std::cout << std::format("[Porth-Util] Thread successfully pinned to core {}\n", core_id);
74 return PorthStatus::SUCCESS;
75}

89[[nodiscard]] inline auto set_realtime_priority() noexcept -> PorthStatus {

93 // SCHED_FIFO: First-In-First-Out real-time scheduling.
