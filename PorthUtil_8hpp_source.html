92[[nodiscard]] inline auto set_realtime_priority() noexcept -> std::expected<void, PorthStatus> {
93 struct sched_param param{};
94 param.sched_priority = MAX_PTHREAD_FIFO_PRIORITY;
95
96 // SCHED_FIFO: First-In-First-Out real-time scheduling.
97 // This prevents the thread from being pushed out of the CPU until it yields.
98 int result = pthread_setschedparam(pthread_self(), SCHED_FIFO, &param);
99
100 if (result != 0) {
101 std::cerr << "[Porth-Util] Warning: Could not set SCHED_FIFO (requires sudo/root)\n";
102 return std::unexpected(PorthStatus::ERROR_PRIORITY);
103 }
104
105 std::cout << "[Porth-Util] Thread priority elevated to Real-Time (SCHED_FIFO)\n";
106 return {};
